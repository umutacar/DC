\section{Problems}

\begin{probl}{Expectation and worst case}
  Explain why if an algorithm has $\Theta(n)$ expected (average) work
  it cannot be the case that once in every $n$ or so tries the
  algorithm requires $\Theta(n^3)$ work.
\end{probl}

\begin{probl}
  Prove the span of an algorithm is at most $\Theta(\log n)$ with
  probability at least $1 - 1/n^5$, and if its worst case span is
  $\Theta(n)$, then its expected span is $\Theta(\log n)$.
\end{exercise}
